Part 1 — Data Preparation
The ML models are only as good as the data they train on. This part covers the complete data preparation pipeline: understanding what data you need, why synthetic data is appropriate for the MVP, how to generate it with realistic behavioural patterns, and how to verify it is correct before training.

1.1 Why Synthetic Data — And Its Honest Limitations
You have no real company login data. You have two options: use a public dataset or generate synthetic data. For this system, synthetic data is the correct choice because public datasets (like CERT Insider Threat) do not contain the behavioural dimensions you need — specifically per-user temporal rhythm, device loyalty scores, and privilege graph traversal patterns.

The limitation you must understand and be ready to explain: the labels in your synthetic data are assigned by rules, and XGBoost learns those rules. This creates a circular dependency — the model learns to reproduce the labeling logic rather than learning from real attacks. This is acceptable for a prototype. In production, HITL decisions from real security analysts replace the synthetic labels over time.

TIP:  When judges ask about data quality, say this: "Our synthetic data bootstraps the model with realistic behavioural patterns. As the system runs in production, HITL gold labels progressively replace synthetic labels in each retraining cycle, eliminating the circularity over 2-3 months."

  python
    # backend/scripts/generate_data.py
    # Run: python scripts/generate_data.py
    # Output: data/sessions.json (1000 sessions, ~8% anomalies)
    
    import json, random, math
    from datetime import datetime, timedelta
    from faker import Faker
    from user_profiles import USERS, RESOURCES, device_id
    
    fake = Faker("en_IN")
    random.seed(42)          # seed for reproducibility
    
    
    # ── HELPER: haversine distance (km) between two lat/lon points ─────
    def haversine(lat1, lon1, lat2, lon2):
        R = 6371  # Earth radius in km
        dlat = math.radians(lat2 - lat1)
        dlon = math.radians(lon2 - lon1)
        a = math.sin(dlat/2)**2 + math.cos(math.radians(lat1)) * \
            math.cos(math.radians(lat2)) * math.sin(dlon/2)**2
        return R * 2 * math.asin(math.sqrt(a))
    
    
    # ── NORMAL SESSION GENERATOR ───────────────────────────────────────
    # This is the baseline. Every user session starts as a normal session.
    # Anomaly injectors then modify specific fields to simulate attacks.
    
    def normal_session(user, base_time, day_offset):
        """Generate one realistic normal login for a given user."""
        
        # Login hour: sample from Gaussian around user average
        hour = int(random.gauss(
            (user["hours"][0] + user["hours"][1]) / 2,  # mean
            user["hour_std"]                             # std deviation
        ))
        hour = max(0, min(23, hour))  # clamp to valid range
        
        # Weekday vs weekend logic
        day = day_offset % 7
        is_weekend = day >= 5
        if is_weekend and random.random() > user["weekend_bias"]:",
            return None  # this user doesnt work weekends — skip this slot
        
        # Pick a resource from their typical set
        resource = random.choice(user["resources"])
        
        # Timestamp: base_time + day_offset days + random minutes in hour
        ts = base_time + timedelta(
            days=day_offset,
            hours=hour,
            minutes=random.randint(0, 59),
            seconds=random.randint(0, 59)
        )
        
        # Failed attempts: mostly 0, occasionally 1 (mistyped password)
        failed = 0
        if random.random() < 0.04:  # 4% of normal logins have 1 failed attempt
            failed = 1
        
        # Device: 90% primary, 10% secondary (e.g., work laptop vs home laptop)
        device_variant = 0
        if user["device_count"] > 1 and random.random() < 0.10:
            device_variant = random.randint(1, user["device_count"] - 1)
        
        return {
            "user_id":             user["id"],
            "role":                user["role"],
            "sensitivity_ceiling": user["sensitivity_ceiling"],
            "timestamp":           ts.isoformat(),
            "hour":                hour,
            "day_of_week":         day,
            "is_weekend":          is_weekend,
            "country":             user["country"],
            "city":                user["city"],
            "lat":                 user["lat"] + random.gauss(0, 0.01),  # small jitter
            "lon":                 user["lon"] + random.gauss(0, 0.01),
            "device_id":           device_id(user["id"], device_variant),
            "primary_device":      device_id(user["id"], 0),
            "resource":            resource,
            "resource_sensitivity": RESOURCES[resource],
            "failed_attempts":     failed,
            "session_duration_s":  int(random.gauss(1800, 600)),  # ~30 min sessions
            "is_anomaly":          False,
            "anomaly_type":        None,
            "risk_label":          "low",   # assigned by assign_label() later
        }
    
    
    # ── SIX ANOMALY INJECTORS ──────────────────────────────────────────
    # Each injector takes a normal session and modifies it to simulate
    # a specific attack type. The is_anomaly flag tells XGBoost this is
    # a high-risk training example.
    
    def inject_impossible_travel(session, user):",
        """Simulates: attacker in another country using stolen credentials."""
        s = session.copy()
        # Pick a city far from the user home city
        foreign_cities = {
            "IN": ("GB", "London", 51.5074, -0.1278),
        }
        country, city, lat, lon = foreign_cities.get(user["country"],
            ("US", "New York", 40.7128, -74.0060))
        
        s["country"]      = country
        s["city"]         = city
        s["lat"]          = lat
        s["lon"]          = lon
        s["is_anomaly"]   = True
        s["anomaly_type"] = "impossible_travel"
        return s
    
    
    def inject_new_device(session, user):
        """Simulates: attacker using their own machine with stolen password."""
        s = session.copy()
        # Device variant 9 is always unknown (never generated in normal sessions)
        s["device_id"]    = device_id(user["id"], 9)
        s["is_anomaly"]   = True
        s["anomaly_type"] = "new_device"
        return s
    
    
    def inject_brute_force(session):
        """Simulates: automated password guessing that eventually succeeds."""
        s = session.copy()
        s["failed_attempts"] = random.randint(6, 15)
        s["is_anomaly"]      = True
        s["anomaly_type"]    = "brute_force"
        return s
    
    
    def inject_unusual_hour(session, user):
        """Simulates: after-hours data exfiltration by insider."""
        s = session.copy()
        # Force login to 2-4am regardless of user normal hours
        s["hour"]         = random.randint(2, 4)
        ts = datetime.fromisoformat(s["timestamp"])
        s["timestamp"]    = ts.replace(hour=s["hour"]).isoformat()
        s["is_anomaly"]   = True
        s["anomaly_type"] = "unusual_hour"
        return s
    
    
    def inject_privilege_escalation(session, user):
        """Simulates: user tries to access resource above their role ceiling."""
        s = session.copy()
        # Find resources above this user's sensitivity ceiling
        escalation_targets = [
            r for r, v in RESOURCES.items()
            if v > user["sensitivity_ceiling"]
        ]
        if not escalation_targets:
            return None  # superadmin has no ceiling — skip
        target = random.choice(escalation_targets)
        s["resource"]             = target
        s["resource_sensitivity"] = RESOURCES[target]
        s["is_anomaly"]           = True
        s["anomaly_type"]         = "privilege_escalation"
        return s
    
    
    def inject_compound_attack(session, user):
        """Simulates: full account takeover — all signals fire at once.
        This is the highest-risk scenario. Combine multiple anomalies.
        """
        # Stack: impossible travel + new device + high-sensitivity resource
        s = inject_impossible_travel(session, user)
        s = inject_new_device(s, user)
        # Force a high-sensitivity resource target
        high_sensitivity = [r for r, v in RESOURCES.items() if v >= 8]
        if high_sensitivity:
            s["resource"]             = random.choice(high_sensitivity)
            s["resource_sensitivity"] = RESOURCES[s["resource"]]
        s["anomaly_type"] = "compound_attack"
        return s
    
    
    # ── LABEL ASSIGNMENT ───────────────────────────────────────────────
    # This function assigns the training label to each session.
    # Labels are integers: 0=low, 1=medium, 2=high.
    # XGBoost learns to classify new sessions into these categories.
    
    def assign_label(session):
        """Deterministic label assignment from session features."""
        atype = session.get("anomaly_type")
        
        # Definite high risk
        if atype in ["impossible_travel", "brute_force", "compound_attack"]:
            return 2  # high
        
        # High risk: privilege escalation to very sensitive resource
        if atype == "privilege_escalation":
            if session["resource_sensitivity"] >= 9:
                return 2  # high
            return 1      # medium — escalation to moderately sensitive resource
        
        # Medium risk: single soft anomaly signals
        if atype in ["new_device", "unusual_hour"]:
            return 1  # medium — suspicious but not confirmed
        
        # Low risk: everything looks normal
        return 0
    
    
    # ── MAIN GENERATION LOOP ───────────────────────────────────────────
    
    def generate_all_sessions(n_days=90, anomaly_rate=0.08):
        """
        Generate n_days of sessions for all users.
        anomaly_rate: fraction of sessions that are anomalous.
        """
        all_sessions = []
        base_time = datetime.now() - timedelta(days=n_days)
        
        anomaly_injectors = [
            inject_impossible_travel,
            inject_new_device,
            inject_brute_force,
            inject_unusual_hour,
            inject_privilege_escalation,
            inject_compound_attack,
        ]
        
        for user in USERS:
            for day_offset in range(n_days):
                # Each user logs in approximately login_freq_per_week / 7 per day
                if random.random() > user["login_freq_per_week"] / 7:
                    continue  # no login this day for this user
                
                # Generate a normal session for this user+day
                session = normal_session(user, base_time, day_offset)
                if session is None:
                    continue  # weekend skip
                
                # Decide: is this an anomaly?
                if random.random() < anomaly_rate:
                    injector = random.choice(anomaly_injectors)
                    injected = injector(session, user) if injector in [
                        inject_impossible_travel,
                        inject_unusual_hour,
                        inject_privilege_escalation,
                        inject_compound_attack,
                    ] else injector(session)
                    if injected is not None:
                        session = injected
                
                # Assign integer label for XGBoost
                session["risk_label"] = assign_label(session)
                all_sessions.append(session)
        
        # Sort chronologically — critical for user history lookups
        all_sessions.sort(key=lambda x: x["timestamp"])
        return all_sessions
    
    
    if __name__ == "__main__":
        import os
        os.makedirs("data", exist_ok=True)
        
        sessions = generate_all_sessions(n_days=90, anomaly_rate=0.08)
        
        with open("data/sessions.json", "w") as f:
            json.dump(sessions, f, indent=2)
        
        # Print verification summary
        total      = len(sessions)
        anomalies  = sum(1 for s in sessions if s["is_anomaly"])
        by_label   = {0:0, 1:0, 2:0}
        by_type    = {}
        for s in sessions:
            by_label[s["risk_label"]] += 1
            if s["anomaly_type"]:
                by_type[s["anomaly_type"]] = by_type.get(s["anomaly_type"], 0) + 1
        
        print(f"Total sessions:   {total}")
        print(f"Anomalies:        {anomalies} ({anomalies/total*100:.1f}%)")
        print(f"Label 0 (low):    {by_label[0]}")
        print(f"Label 1 (medium): {by_label[1]}")
        print(f"Label 2 (high):   {by_label[2]}")
        print(f"Anomaly types:    {by_type}")
        print("Saved to data/sessions.json")

Part 2 — Feature Engineering
Raw session data cannot be fed to ML models directly. Feature engineering transforms raw fields (hour=14, country="IN") into numerical signals that encode behavioural meaning (hour_zscore=2.1 means this hour is 2.1 standard deviations unusual for this user). This is the most important step — the model is only as smart as the signals you give it.

2.1 The 10 Features — Explained in Detail

2.2 Complete features.py — Full Working Code
  python
    # backend/features.py
    # Import: from features import engineer_features
    
    import math
    import numpy as np
    from datetime import datetime
    from user_profiles import RESOURCES
    
    
    def engineer_features(event: dict, user_history: list) -> dict:
        """
        Convert one raw login event into a 10-feature vector.
        
        Args:
            event:        dict with raw login fields
            user_history: list of prior session dicts for this user,
                          sorted chronologically. Can be empty for new users.
        
        Returns:
            dict with exactly 10 keys, all float or int. No None values.
        
        CRITICAL: The order of keys in the returned dict must match
        the column order used during training. Never add, remove,
        or reorder features without retraining all models.
        """
        hour = event.get("hour", 12)  # default to noon for missing data
        
        # ── FEATURE 1 & 2: Cyclical hour encoding ─────────────────────
        # WHY: Raw hour (0-23) treats 23 and 0 as far apart.
        # Sin/cos encoding makes them adjacent on the unit circle.
        hour_sin = math.sin(2 * math.pi * hour / 24)
        hour_cos = math.cos(2 * math.pi * hour / 24)
        
        # ── FEATURE 3: Z-score of login hour vs user history ───────────
        # WHY: Measures how unusual THIS hour is for THIS specific user.
        # Alice always logs in at 10am — a 3am login is highly unusual.
        # Dave logs in at all hours — a 3am login is normal for him.
        if len(user_history) >= 5:
            user_hours = [s["hour"] for s in user_history]
            hour_mean  = np.mean(user_hours)
            hour_std   = np.std(user_hours)
        else:
            hour_mean, hour_std = 12.0, 4.0  # neutral defaults for new users
        hour_zscore = (hour - hour_mean) / (hour_std + 0.1)  # +0.1 prevents div/0
        
        # ── FEATURE 4: New device flag ─────────────────────────────────
        # WHY: Attackers use their own machine. Device fingerprint is
        # something they cannot easily replicate.
        current_device = event.get("device_id", "")
        if user_history:
            known_devices = set(s["device_id"] for s in user_history)
            is_new_device = int(current_device not in known_devices)
        else:
            is_new_device = 0  # no history = cannot say it is new
        
        # ── FEATURE 5: Country change flag ─────────────────────────────
        # WHY: Catches credential use from unexpected locations.
        # Looks at last 5 sessions to handle legitimate travel.
        current_country = event.get("country", "")
        if user_history:
            recent_countries = set(s["country"] for s in user_history[-5:])
            country_change = int(current_country not in recent_countries)
        else:
            country_change = 0
        
        # ── FEATURE 6: Impossible travel flag ──────────────────────────
        # WHY: If country changed AND last login was within 24 hours,
        # the implied travel speed is physically impossible.
        impossible_travel = 0
        if country_change and user_history:
            last_ts = datetime.fromisoformat(user_history[-1]["timestamp"])
            current_ts = datetime.fromisoformat(event["timestamp"])
            hours_elapsed = (current_ts - last_ts).total_seconds() / 3600
            if hours_elapsed < 6:   # less than 6 hours between countries
                impossible_travel = 1
        
        # ── FEATURE 7: Failed login attempts ───────────────────────────
        # WHY: Multiple failures before success = likely brute force.
        # Capped at 15 to prevent extreme outliers dominating the model.
        failed_attempts = min(event.get("failed_attempts", 0), 15)
        
        # ── FEATURE 8: Resource sensitivity ────────────────────────────
        # WHY: Scales all other risk signals by the value of what
        # is being accessed. Small anomaly + high sensitivity = high risk.
        resource = event.get("resource", "")
        resource_sensitivity = event.get(
            "resource_sensitivity",
            RESOURCES.get(resource, 3)  # default to 3 if unknown resource
        )
        
        # ── FEATURE 9: Days since last login ───────────────────────────
        # WHY: Dormant accounts suddenly used can indicate compromise.
        # Also: very frequent logins can indicate automated/bot activity.
        if user_history:
            last_ts = datetime.fromisoformat(user_history[-1]["timestamp"])
            current_ts = datetime.fromisoformat(event["timestamp"])
            days_since = max(0, (current_ts - last_ts).days)
            days_since = min(days_since, 90)  # cap at 90 days
        else:
            days_since = 30  # neutral default for first-ever login
        
        # ── FEATURE 10: Privilege gap score ────────────────────────────
        # WHY: Catches insider threats and privilege escalation.
        # Even if all other signals are clean, accessing a resource
        # far above your role ceiling is suspicious.
        role_ceiling = event.get("sensitivity_ceiling", 5)
        gap = resource_sensitivity - role_ceiling
        privilege_gap_score = max(0.0, min(gap / 10.0, 1.0))
        
        # ── RETURN: Fixed-order dict (ORDER IS CRITICAL) ───────────────
        # This exact order must match what train_models.py expects.
        # Never change the order of keys without retraining.
        return {
            "hour_sin":            round(float(hour_sin), 6),
            "hour_cos":            round(float(hour_cos), 6),
            "hour_zscore":         round(float(hour_zscore), 6),
            "is_new_device":       int(is_new_device),
            "country_change":      int(country_change),
            "impossible_travel":   int(impossible_travel),
            "failed_attempts":     int(failed_attempts),
            "resource_sensitivity":int(resource_sensitivity),

Part 3 — Model Training
This part covers the complete model training pipeline. You will train three things: a StandardScaler (data preprocessing), an Isolation Forest (anomaly detection), and an XGBoost classifier (risk classification). All three are saved as .pkl files and loaded together at FastAPI startup.

3.1 Why You Must Scale Features First
StandardScaler transforms all features so each has mean=0 and standard deviation=1. Without this, features with large ranges (resource_sensitivity: 1–10) will dominate features with small ranges (hour_sin: −1 to +1), making the model ignore the smaller features.

TIP:  Fit the scaler ONCE on the training data. Never fit it again. During inference, call scaler.transform() only — not fit_transform(). Calling fit again at inference time would recalculate the mean and std from the inference data, breaking all past predictions.

3.3 Understanding XGBoost Class Imbalance
Your dataset has ~870 low-risk sessions and ~60 high-risk sessions. Without correction, XGBoost learns to always predict "low risk" and achieves 87% accuracy while being completely useless. The scale_pos_weight parameter fixes this by telling XGBoost that high-risk examples are more important.

  python
    # How to calculate scale_pos_weight:
    n_low  = sum(1 for s in sessions if s["risk_label"] == 0)   # e.g., 870
    n_high = sum(1 for s in sessions if s["risk_label"] == 2)   # e.g., 35
    scale_pos_weight = n_low / n_high  # e.g., 870/35 = 24.8 → use 20-25
    
    # Rule of thumb: if ratio is 10:1, use scale_pos_weight=10
    #                if ratio is 25:1, use scale_pos_weight=20 (slightly lower)
    #                if ratio is 50:1, use scale_pos_weight=30 (cap here)

    def build_feature_matrix(sessions):
        """
        Convert list of raw sessions into numpy arrays X and y.
        Processes sessions in chronological order so user_history
        is always built from past events only — not future ones.
        This simulates real inference conditions during training.
        """
        X_rows   = []
        y_labels = []
        user_histories = {}  # {user_id: [sessions seen so far]}
        
        for session in sessions:  # already sorted chronologically
            uid = session["user_id"]
            history = user_histories.get(uid, [])
            
            # Compute features using only past history
            feats = engineer_features(session, history)
            X_rows.append(list(feats.values()))  # preserves dict order
            y_labels.append(session["risk_label"])
            
            # Update history AFTER computing features
            # (we only know about sessions that already happened)
            user_histories.setdefault(uid, []).append(session)
        
        return np.array(X_rows, dtype=np.float32), np.array(y_labels, dtype=np.int32)
    
    
    def train_and_save():
        print(f"[{datetime.now():%H:%M:%S}] Loading session data...")
        with open("data/sessions.json") as f:
            sessions = json.load(f)
        print(f"  Loaded {len(sessions)} sessions")
        
        # ── STEP 1: Build feature matrix ─────────────────────────────
        print(f"[{datetime.now():%H:%M:%S}] Engineering features...")
        X, y = build_feature_matrix(sessions)
        print(f"  Feature matrix shape: {X.shape}")
        print(f"  Label distribution: {dict(Counter(y))}")
        
        # ── STEP 2: Split into train / test ──────────────────────────
        # stratify=y ensures both splits have all 3 label classes
        X_train, X_test, y_train, y_test = train_test_split(
            X, y,
            test_size=0.20,      # 20% held out for evaluation
            random_state=42,
            stratify=y           # CRITICAL: preserves class ratios
        )
        print(f"  Train: {len(X_train)}, Test: {len(X_test)}")
        
        # ── STEP 3: Fit and apply StandardScaler ─────────────────────
        print(f"[{datetime.now():%H:%M:%S}] Fitting StandardScaler...")
        scaler = StandardScaler()
        X_train_scaled = scaler.fit_transform(X_train)  # FIT only on train
        X_test_scaled  = scaler.transform(X_test)        # transform only on test
        print(f"  Scaler fitted. Feature means: {scaler.mean_.round(3)}")
        
        # ── STEP 4: Train Isolation Forest ───────────────────────────
        print(f"[{datetime.now():%H:%M:%S}] Training Isolation Forest...")
        n_anomalies = sum(1 for s in sessions if s["is_anomaly"])
        contamination = n_anomalies / len(sessions)
        print(f"  Contamination set to: {contamination:.3f} ({n_anomalies} anomalies)")
        
        iso_forest = IsolationForest(
            n_estimators=150,       # more trees = more stable scores
            max_samples="auto",     # subsample per tree (auto = min(256, n))
            contamination=contamination,
            max_features=1.0,       # use all features per tree
            bootstrap=False,        # sample without replacement
            random_state=42,
            n_jobs=-1,              # use all CPU cores
        )
        # Fit on ALL training data — Isolation Forest is unsupervised
        # (does not use y_train labels)
        iso_forest.fit(X_train_scaled)
        
        # Evaluate: anomaly scores on test set
        # decision_function: negative = anomaly, positive = normal
        iso_scores = iso_forest.decision_function(X_test_scaled)
        iso_pred   = iso_forest.predict(X_test_scaled)  # -1=anomaly, 1=normal
        
        # Convert to our convention: 1=anomaly, 0=normal
        iso_binary = (iso_pred == -1).astype(int)
        # True anomaly = label is 1 or 2
        true_anomaly = (y_test > 0).astype(int)
        correct = (iso_binary == true_anomaly).mean()
        print(f"  Isolation Forest anomaly detection accuracy: {correct:.3f}")
        
        # ── STEP 5: Calculate XGBoost class weights ───────────────────
        label_counts = Counter(y_train)
        n_majority = label_counts[0]   # low risk — most common
        n_minority = label_counts[2]   # high risk — least common
        if n_minority > 0:
            spw = min(n_majority / n_minority, 30)  # cap at 30
        else:
            spw = 10  # fallback
        print(f"  Class counts: {dict(label_counts)}")
        print(f"  scale_pos_weight: {spw:.1f}")
        
        # ── STEP 6: Train XGBoost ─────────────────────────────────────
        print(f"[{datetime.now():%H:%M:%S}] Training XGBoost classifier...")
        xgb_model = XGBClassifier(
            n_estimators=200,           # number of boosting rounds
            max_depth=4,                # tree depth — 4-6 ideal for tabular
            learning_rate=0.1,          # shrinkage — prevents overfitting
            subsample=0.8,              # row subsampling per tree
            colsample_bytree=0.8,       # feature subsampling per tree
            scale_pos_weight=spw,       # class imbalance correction
            use_label_encoder=False,
            eval_metric="mlogloss",     # multi-class log loss
            objective="multi:softprob", # outputs probabilities
            num_class=3,                # low / medium / high
            random_state=42,
            n_jobs=-1,
        )
        
        # Early stopping: stop if validation loss doesnt improve for 20 rounds
        xgb_model.fit(
            X_train_scaled, y_train,
            eval_set=[(X_test_scaled, y_test)],
            verbose=False,
        )
        print(f"  XGBoost trained. Best iteration: {xgb_model.best_iteration}")
        
        # ── STEP 7: Evaluate XGBoost ──────────────────────────────────
        y_pred = xgb_model.predict(X_test_scaled)
        accuracy = (y_pred == y_test).mean()
        print(f"  XGBoost overall accuracy: {accuracy:.3f}")
        print()
        print("  Classification Report:")
        print(classification_report(
            y_test, y_pred,
            target_names=["low","medium","high"],
            zero_division=0
        ))
        
        # ── STEP 8: Setup SHAP explainer ──────────────────────────────
        print(f"[{datetime.now():%H:%M:%S}] Setting up SHAP explainer...")
        explainer = shap.TreeExplainer(xgb_model)
        # Quick test: compute SHAP values for first 5 test samples
        sample_shap = explainer.shap_values(X_test_scaled[:5])
        print(f"  SHAP explainer OK. Output shape: {np.array(sample_shap).shape}")
        
        # ── STEP 9: Save all models ───────────────────────────────────
        print(f"[{datetime.now():%H:%M:%S}] Saving models...")
        os.makedirs("models", exist_ok=True)
        
        joblib.dump(iso_forest, "models/isolation_forest.pkl")
        joblib.dump(xgb_model,  "models/xgboost.pkl")
        joblib.dump(scaler,     "models/scaler.pkl")
        
        # Also save metadata for the retrain pipeline
        metadata = {
            "trained_at":      datetime.now().isoformat(),
            "n_train_samples": len(X_train),
            "contamination":   contamination,
            "scale_pos_weight":spw,
            "xgb_accuracy":    round(float(accuracy), 4),
            "feature_names":   FEATURE_NAMES,
            "label_counts":    {str(k):v for k,v in label_counts.items()},
        }
        with open("models/metadata.json", "w") as f:
            json.dump(metadata, f, indent=2)
        
        print("  Saved: models/isolation_forest.pkl")
        print("  Saved: models/xgboost.pkl")
        print("  Saved: models/scaler.pkl")
        print("  Saved: models/metadata.json")
        print()
        print("=== TRAINING COMPLETE ===")
        print(f"XGBoost accuracy: {accuracy:.3f}")
        if accuracy < 0.80:
            print("WARNING: Accuracy below 0.80. Check your data and labels.")
        else:
            print("Models are ready for use in the FastAPI backend.")
    
    
    if __name__ == "__main__":
        train_and_save()

4.2 The Inference Pipeline — Full Code
  python
    # backend/scorer.py
    # Called by POST /evaluate-login on every login event.
    
    import numpy as np
    from features import engineer_features
    from dna_engine import build_dna, dna_match_score, detect_drift
    
    
    def score_login_event(event: dict, user_history: list, app_state) -> dict:
        """
        Full scoring pipeline for one login event.
        Returns complete risk assessment dict.
        """
        # ── STEP 1: Engineer features ────────────────────────────────
        feats = engineer_features(event, user_history)
        X = np.array([list(feats.values())], dtype=np.float32)
        
        # ── STEP 2: Scale features ────────────────────────────────────
        # Use transform() ONLY — never fit_transform() at inference time
        X_scaled = app_state.scaler.transform(X)
        
        # ── STEP 3: Isolation Forest score ───────────────────────────
        # decision_function returns float: negative = anomalous
        iso_score_raw = float(app_state.iso.decision_function(X_scaled)[0])
        
        # Normalize to 0-1: more anomalous = higher value
        anomaly_norm = max(0.0, min(1.0, 0.5 - iso_score_raw * 0.5))
        
        # ── STEP 4: XGBoost risk probability ─────────────────────────
        # predict_proba returns [prob_low, prob_medium, prob_high]
        proba = app_state.xgb.predict_proba(X_scaled)[0]
        xgb_proba_high = float(proba[2])  # probability of HIGH risk
        xgb_proba_med  = float(proba[1])
        
        # ── STEP 5: SHAP feature contributions ───────────────────────
        # shap_values shape: (n_classes, n_samples, n_features)
        shap_values = app_state.explainer.shap_values(X_scaled)
        # Get contributions for the HIGH risk class (index 2)
        shap_for_high = shap_values[2][0]  # shape: (10,)
        
        # Pair feature names with their SHAP contributions
        from features import FEATURE_NAMES, FEATURE_LABELS
        shap_pairs = sorted(
            zip(FEATURE_NAMES, shap_for_high.tolist()),
            key=lambda x: abs(x[1]),
            reverse=True
        )
        top_shap = [
            {"feature": FEATURE_LABELS[f], "impact": round(v, 4)}
            for f, v in shap_pairs[:3]
        ]
        
        # ── STEP 6: DNA match score ───────────────────────────────────
        uid = event["user_id"]
        dna = app_state.dna_cache.get(uid)  # None if new user
        dna_match = dna_match_score(dna, event)  # 0.0 to 1.0
        dna_conf  = dna["confidence"] if dna else 0.05
        drift_status = detect_drift(dna, event) if dna else "BUILDING"
        
        # ── STEP 7: Privilege gap ─────────────────────────────────────
        graph_penalty = float(feats["privilege_gap_score"])
        
        # ── STEP 8: Combined risk score ───────────────────────────────
        dna_weight  = 0.10 + (dna_conf * 0.30)   # 0.10 → 0.40
        xgb_weight  = 0.40 - (dna_conf * 0.10)   # 0.40 → 0.30
        ml_weight   = 0.40                        # Isolation Forest fixed
        graph_weight = 0.20                       # Graph fixed
        dna_penalty = (1.0 - dna_match) * dna_weight
        
        raw_score = (
            ml_weight    * anomaly_norm  +
            xgb_weight   * xgb_proba_high +
            dna_penalty  +
            graph_weight * graph_penalty
        )
        risk_score = round(min(raw_score * 100, 100.0), 1)
        
        # ── STEP 9: Access decision ───────────────────────────────────
        if risk_score < 30:   decision = "ALLOW"
        elif risk_score < 60: decision = "MFA"
        elif risk_score < 80: decision = "STEP_UP"
        else:                 decision = "BLOCK"
        
        return {
            "risk_score":      risk_score,
            "decision":        decision,
            "iso_score_raw":   round(iso_score_raw, 4),
            "anomaly_norm":    round(anomaly_norm, 4),
            "xgb_proba_high":  round(xgb_proba_high, 4),
            "dna_match":       round(dna_match, 4),
            "dna_confidence":  round(dna_conf, 4),
            "drift_status":    drift_status,
            "graph_penalty":   round(graph_penalty, 4),
            "shap_top_3":      top_shap,
            "feature_vector":  feats,
        }

    # backend/retrain.py
    # Called automatically by APScheduler every Sunday at 2am,

    def run_retrain_pipeline(force=False) -> dict:
        """
        Full retraining pipeline.
        Returns dict with status and accuracy comparison.
        force=True: retrain even if new data is sparse.
        """
        db = SessionLocal()
        print(f"[{datetime.now():%H:%M:%S}] Starting retrain pipeline...")
        
        # ── STEP 1: Count available new HITL labels ───────────────────
        new_hitl = db.query(HITLDecision).filter(
            HITLDecision.used_in_retrain == False
        ).all()
        
        print(f"  New HITL decisions available: {len(new_hitl)}")
        
        if not force and len(new_hitl) < 10:
            db.close()
            return {"status": "skipped", "reason": "< 10 new HITL decisions"}
        
        # ── STEP 2: Load base synthetic data ──────────────────────────
        with open("data/sessions.json") as f:
            base_sessions = json.load(f)
        print(f"  Base synthetic sessions: {len(base_sessions)}")
        
        # ── STEP 3: Load real events from last 60 days ────────────────
        cutoff = datetime.utcnow() - timedelta(days=60)
        real_events = db.query(LoginEvent).filter(
            LoginEvent.timestamp >= cutoff,
            LoginEvent.decision != None
        ).all()
        print(f"  Real events (last 60 days): {len(real_events)}")
        
        # ── STEP 4: Build combined feature matrix with weights ─────────
        X_all, y_all, w_all = [], [], []
        label_map = {"ALLOW": 0, "MFA": 1, "STEP_UP": 1, "BLOCK": 2}
        user_histories = {}
        
        def add_session(session_dict, label_int, weight):
            uid = session_dict.get("user_id", "unknown")
            history = user_histories.get(uid, [])
            feats = engineer_features(session_dict, history)
            X_all.append(list(feats.values()))
            y_all.append(label_int)
            w_all.append(weight)
            user_histories.setdefault(uid, []).append(session_dict)
        
        # Add synthetic data (weight 1.0)
        for s in sorted(base_sessions, key=lambda x: x["timestamp"]):
            add_session(s, s["risk_label"], 1.0)
        
        # Add real events (weight 2.0)
        for e in real_events:
            s = {
                "user_id": e.user_id, "hour": e.hour,
                "country": e.country, "device_id": e.device_id,
                "resource": e.resource,
                "resource_sensitivity": e.resource_sensitivity,
                "sensitivity_ceiling": 5,  # default if not stored
                "failed_attempts": e.failed_attempts,
                "timestamp": e.timestamp.isoformat(),
            }
            label = label_map.get(e.decision, 0)
            add_session(s, label, 2.0)
        
        # Add HITL decisions (weight 3.0 — gold labels)
        for h in new_hitl:
            event = db.query(LoginEvent).filter(
                LoginEvent.id == h.event_id
            ).first()
            if not event:
                continue
            s = {
                "user_id": event.user_id, "hour": event.hour,
                "country": event.country, "device_id": event.device_id,
                "resource": event.resource,
                "resource_sensitivity": event.resource_sensitivity,
                "sensitivity_ceiling": 5,
                "failed_attempts": event.failed_attempts,
                "timestamp": event.timestamp.isoformat(),
            }
            label = label_map.get(h.admin_decision, 0)
            add_session(s, label, 3.0)
        
        X = np.array(X_all, dtype=np.float32)
        y = np.array(y_all, dtype=np.int32)
        w = np.array(w_all, dtype=np.float32)
        print(f"  Total training samples: {len(X)}")
        print(f"  Label distribution: {dict(Counter(y))}")
        
        # ── STEP 5: Train/test split ──────────────────────────────────
        X_tr, X_te, y_tr, y_te, w_tr, _ = train_test_split(
            X, y, w, test_size=0.20, random_state=42, stratify=y
        )
        
        # ── STEP 6: Fit new scaler and train new models ───────────────
        new_scaler = StandardScaler()
        X_tr_s = new_scaler.fit_transform(X_tr)
        X_te_s = new_scaler.transform(X_te)
        
        n_maj = sum(y_tr == 0)
        n_min = max(sum(y_tr == 2), 1)
        spw = min(n_maj / n_min, 30)
        
        new_iso = IsolationForest(n_estimators=150, contamination=0.08,
                                 random_state=42, n_jobs=-1)
        new_iso.fit(X_tr_s)
        
        new_xgb = XGBClassifier(
            n_estimators=200, max_depth=4, learning_rate=0.1,
            subsample=0.8, colsample_bytree=0.8,
            scale_pos_weight=spw, use_label_encoder=False,
            eval_metric="mlogloss", objective="multi:softprob",
            num_class=3, random_state=42, n_jobs=-1,
        )
        new_xgb.fit(X_tr_s, y_tr, sample_weight=w_tr,
                    eval_set=[(X_te_s, y_te)], verbose=False)
        
        # ── STEP 7: Compare new vs old model ──────────────────────────
        new_acc = accuracy_score(y_te, new_xgb.predict(X_te_s))
        
        old_xgb    = joblib.load("models/xgboost.pkl")
        old_scaler = joblib.load("models/scaler.pkl")
        old_acc    = accuracy_score(y_te, old_xgb.predict(old_scaler.transform(X_te)))
        
        print(f"  Old model accuracy: {old_acc:.3f}")
        print(f"  New model accuracy: {new_acc:.3f}")
        
        # ── STEP 8: Replace models only if new is not significantly worse
        REGRESSION_TOLERANCE = 0.03  # allow up to 3% regression
        if new_acc >= old_acc - REGRESSION_TOLERANCE:
            joblib.dump(new_iso,    "models/isolation_forest.pkl")
            joblib.dump(new_xgb,    "models/xgboost.pkl")
            joblib.dump(new_scaler, "models/scaler.pkl")
            
            # Mark HITL decisions as used
            for h in new_hitl:
                h.used_in_retrain = True
            db.commit()
            db.close()
            
            print("  Models updated successfully.")
            return {
                "status": "updated",
                "old_accuracy": round(old_acc, 4),
                "new_accuracy": round(new_acc, 4),
                "hitl_labels_used": len(new_hitl),
            }
        else:
            db.close()
            print("  New model is significantly worse — keeping old models.")
            return {
                "status": "rejected",
                "old_accuracy": round(old_acc, 4),
                "new_accuracy": round(new_acc, 4),
                "reason": f"regression of {old_acc - new_acc:.3f} exceeds tolerance",
            }
    
    
    if __name__ == "__main__":
        result = run_retrain_pipeline(force=True)
        print(result)

    from apscheduler.schedulers.background import BackgroundScheduler
    from retrain import run_retrain_pipeline

    def check_and_retrain():
        """Called every hour. Retrain if enough HITL data has arrived."""
        db = SessionLocal()
        pending = db.query(HITLDecision).filter(
            HITLDecision.used_in_retrain == False
        ).count()
        db.close()
        
        if pending >= 50:
            print(f"[Scheduler] {pending} HITL decisions pending — retraining")
            result = run_retrain_pipeline()
            print(f"[Scheduler] Retrain result: {result}")
    
    scheduler = BackgroundScheduler()
    scheduler.add_job(check_and_retrain, "interval", hours=1)
    scheduler.add_job(
        lambda: run_retrain_pipeline(force=True),
        "cron", day_of_week="sun", hour=2
    )
    scheduler.start()
